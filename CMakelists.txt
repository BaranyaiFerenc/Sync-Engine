cmake_minimum_required(VERSION 3.10)

# Set the project name
project(Sync-Engine)

# Add executable
add_executable(Sync-Engine main.cpp dictionary.hpp list.hpp matrix.hpp string.hpp)

# Enable all compiler warnings
if (CMAKE_CXX_COMPILER_ID STREQUAL "GNU" OR CMAKE_CXX_COMPILER_ID STREQUAL "Clang")
    target_compile_options(Sync-Engine PRIVATE -Wall)
elseif (CMAKE_CXX_COMPILER_ID STREQUAL "MSVC")
    target_compile_options(Sync-Engine PRIVATE /W4)
endif()
