cmake_minimum_required(VERSION 3.10)

# Set the project name
project(sync_engine)

# Add executable
add_executable(sync_engine main.cpp dictionary.hpp list.hpp matrix.hpp string.hpp)

# Enable all compiler warnings
if (CMAKE_CXX_COMPILER_ID STREQUAL "GNU" OR CMAKE_CXX_COMPILER_ID STREQUAL "Clang")
    target_compile_options(sync_engine PRIVATE -Wall)
elseif (CMAKE_CXX_COMPILER_ID STREQUAL "MSVC")
    target_compile_options(sync_engine PRIVATE /W4)
endif()
